import { buildDynamicMDX, buildDynamicMeta } from 'nextra/remote'
import { RemoteContent } from 'nextra/data'
import { promises as fs } from 'fs'
import path from 'path'

export const getStaticProps = async ({ params: { table } }) => {
  const tableFile = path.join(process.cwd(), `tables/github/${table}.md`)
  const tableContent = await fs.readFile(tableFile, 'utf8')
  return {
    props: {
      ...(await buildDynamicMDX(tableContent)),
      ...(await buildDynamicMeta())
    },
  }
}

export const getStaticPaths = () => {
  return {
    paths: [],
    fallback: 'blocking'
  }
}
